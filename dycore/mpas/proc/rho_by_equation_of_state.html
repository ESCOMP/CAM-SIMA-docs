    pure elemental function rho_by_equation_of_state(constant_r, p, t) result(rho)
        use, intrinsic :: iso_fortran_env, only: real64

        real(real64), intent(in) :: constant_r, p, t
        real(real64) :: rho

        rho = p / (constant_r * t)
    end function rho_by_equation_of_state
