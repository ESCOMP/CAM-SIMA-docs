    pure elemental function dzw_of_rdzw_real32(rdzw) result(dzw)
        use, intrinsic :: iso_fortran_env, only: real32

        real(real32), intent(in) :: rdzw
        real(real32) :: dzw

        dzw = 1.0_real32 / rdzw
    end function dzw_of_rdzw_real32
