    pure function reverse(array)
        use, intrinsic :: iso_fortran_env, only: real64

        real(real64), intent(in) :: array(:)
        real(real64) :: reverse(size(array))

        integer :: n

        n = size(array)

        ! There is nothing to reverse.
        if (n == 0) then
            return
        end if

        reverse(:) = array(n:1:-1)
    end function reverse
